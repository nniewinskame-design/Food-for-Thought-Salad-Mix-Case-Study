    <script>
        let flippedCards = 0;
        const totalCards = 15; // 5 questions Ã— 3 cards each

        function flipCard(card, type) {
            if (card.classList.contains('flipped')) return;
            
            card.classList.add('flipped');
            flippedCards++;
            
            // Update progress bar
            const progress = (flippedCards / totalCards) * 100;
            document.getElementById('progressFill').style.width = progress + '%';
            
            // Show completion message when all cards are flipped
            if (flippedCards === totalCards) {
                setTimeout(() => {
                    document.getElementById('completionMessage').style.display = 'block';
                    document.getElementById('completionMessage').scrollIntoView({ behavior: 'smooth' });
                }, 1000);
            }
        }

        function resetGame() {
            // Reset all cards
            const cards = document.querySelectorAll('.flashcard');
            cards.forEach(card => {
                card.classList.remove('flipped');
            });
            
            // Reset progress
            flippedCards = 0;
            document.getElementById('progressFill').style.width = '0%';
            
            // Hide completion message
            document.getElementById('completionMessage').style.display = 'none';
            
            // Scroll to top
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }
    </script>
